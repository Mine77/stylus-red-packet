sol_storage! {
  /// Persistent storage state
  #[entrypoint]
  pub struct Escrow {
    address owner;
    address verifier; // WebAuthn verifier contract
    uint256 claim_amount;
    mapping(bytes32 => bytes32) pubkey_hash; // Prevents replay
    mapping(bytes32 => uint256) balances; // Ledger
  }
}

#[public]
impl Escrow {
  /// Initialize contract with WebAuthn verifier and claim amount
  pub fn initialize(&mut self, verifier: Address, amount: U256) -> bool {
    if self.initialized.get() { return false; }
    self.verifier.set(verifier);
    self.claim_amount.set(amount);
    self.initialized.set(true);
    true
  }

  /// Claim funds by verifying a passkey assertion (One-Time Claim)
  pub fn claim(&mut self, id: Bytes, auth: Bytes, sig: Bytes, ... ) -> bool {
    // 1. Call external WebAuthn verifier contract to validate signature
    if !self.verify_with_contract(auth, sig, ...) { return false; }

    // 2. Ensure this passkey hasn't claimed before (Sybil Resistance)
    let key = self.credential_key(&id);
    if self.pubkey_hash.get(key) != B256::ZERO { return false; }

    // 3. Mark as used and credit balance
    self.pubkey_hash.insert(key, ...);
    self.balances.insert(key, self.claim_amount.get());
    true
  }

  /// Withdraw funds to a recipient address
  pub fn withdraw(&mut self, id: Bytes, recipient: Address, ...) -> bool {
    // 1. Re-verify ownership via Passkey
    if !self.verify_with_contract(...) { return false; }

    // 2. Check balance and transfer ETH via Stylus SDK
    let key = self.credential_key(&id);
    let bal = self.balances.get(key);
    self.balances.insert(key, U256::ZERO); // Optimistic update

    if transfer_eth(self.vm(), recipient, bal).is_err() {
      self.balances.insert(key, bal); // Revert on failure
      return false;
    }
    true
  }
}

Composability

Delegates WebAuthn logic to external verifier.

Sybil Resistance

One-time claim enforced per passkey credential.

Native ETH Transfer

Direct ETH transfers via Stylus SDK methods.
