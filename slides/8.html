//! Stylus WebAuthn verifier using Rust's p256 crate. use p256::ecdsa::{signature::hazmat::PrehashVerifier, Signature, VerifyingKey}; use stylus_sdk::prelude::*; sol_storage! { /// Persistent storage for the WebAuthn verifier. pub struct WebAuthnVerifier { bool initialized; bytes32 rp_id_hash; // Hash of the Relaying Party ID (e.g. "example.com") } } #[public] impl WebAuthnVerifier { /// Initializes the contract with the expected RP ID hash. pub fn initialize(&mut self, rp_id_hash: B256) -> bool { if self.initialized.get() { return false; } self.rp_id_hash.set(rp_id_hash); self.initialized.set(true); true } /// Verifies a WebAuthn assertion signature (P-256). pub fn verify( &self, authenticator_data: Bytes, signed_message_hash: B256, signature: Bytes, pubkey_x: B256, pubkey_y: B256, ) -> bool { let auth_bytes = authenticator_data.as_ref(); // 1. Validate RP ID hash matches storage (prevents phishing) if B256::from_slice(&auth_bytes[0..32]) != self.rp_id_hash.get() { return false; } // 2. Parse public key & signature using Rust's p256 crate let key = VerifyingKey::from_sec1_bytes(&Self::sec1_uncompressed(pubkey_x, pubkey_y)) .ok()?; let sig = Signature::from_slice(signature.as_ref()).ok()?; // 3. Verify signature against the hash key.verify_prehash(signed_message_hash.as_slice(), &sig).is_ok() } }

Standard Crates

Uses p256 crate directly, no custom assembly needed.

Phishing Protection

Verifies RP ID hash on-chain to prevent replay.

Efficient Verify

Pre-hashing off-chain keeps gas costs minimal.
